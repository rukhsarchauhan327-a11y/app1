    <script>
        // Auto calculate price per bottle and new total
        function updateCalculations() {
            const bottles = document.getElementById('bottles-input').value || 0;
            const purchasePrice = document.getElementById('purchase-price').value || 0;
            const currentStock = 24;
            
            const pricePerBottle = bottles > 0 ? (purchasePrice / bottles).toFixed(0) : 0;
            const newTotal = parseInt(currentStock) + parseInt(bottles);
            
            document.getElementById('price-per-bottle').textContent = `₹${pricePerBottle}`;
            document.getElementById('new-total').textContent = `${newTotal} Bottles`;
        }

        // Add event listeners
        document.getElementById('bottles-input').addEventListener('input', updateCalculations);
        document.getElementById('purchase-price').addEventListener('input', updateCalculations);
    </script>
