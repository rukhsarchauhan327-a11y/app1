    <script>
        const quantityTypeRadios = document.querySelectorAll('input[name="quantity_type"]');
        const pricingSection = document.getElementById('pricing-section');
        const weightFields = document.getElementById('weight-fields');
        const unitFields = document.getElementById('unit-fields');
        const unitDropdown = document.getElementById('unit-dropdown');
        const thresholdSlider = document.getElementById('threshold-slider');
        const thresholdValue = document.getElementById('threshold-value');
        const unitLabel = document.getElementById('unit-label');
        const unitTypeDisplay = document.getElementById('unit-type-display');
        const sellingUnitDisplay = document.getElementById('selling-unit-display');
        const weightUnitDisplay = document.getElementById('weight-unit-display');

        quantityTypeRadios.forEach(radio => {
            radio.addEventListener('change', function() {
                if (this.checked) {
                    pricingSection.classList.remove('hidden');
                    
                    if (this.value === 'kg_gram') {
                        weightFields.classList.remove('hidden');
                        unitFields.classList.add('hidden');
                        unitDropdown.innerHTML = `
                            <option value="kg">Kg</option>
                            <option value="gram">Gram</option>
                        `;
                        weightUnitDisplay.textContent = 'Kg';
                    } else if (this.value === 'ltr_ml') {
                        weightFields.classList.remove('hidden');
                        unitFields.classList.add('hidden');
                        unitDropdown.innerHTML = `
                            <option value="ltr">Litre</option>
                            <option value="ml">ml</option>
                        `;
                        weightUnitDisplay.textContent = 'Litre';
                    } else {
                        weightFields.classList.add('hidden');
                        unitFields.classList.remove('hidden');
                        
                        // Update unit labels based on selection
                        const unitType = this.value.charAt(0).toUpperCase() + this.value.slice(1);
                        unitLabel.textContent = `Total ${unitType}s`;
                        unitTypeDisplay.textContent = unitType;
                        sellingUnitDisplay.textContent = unitType;
                    }
                }
            });
        });

        // Update weight unit display when dropdown changes
        unitDropdown.addEventListener('change', function() {
            const selectedUnit = this.value.charAt(0).toUpperCase() + this.value.slice(1);
            weightUnitDisplay.textContent = selectedUnit;
        });

        thresholdSlider.addEventListener('input', function() {
            thresholdValue.textContent = this.value + '%';
        });
    </script>
